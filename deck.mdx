## reduce
```
const numbs = [1, 2, 3, 4, 5, 6, 7];

//filter with reduce
const filtered = numbs.reduce((result, curr) => {
  if (curr < 3) {
    result.push(curr);
  }
  return result;
}, []);

//map with reduce
const mapped = numbs.reduce((result, curr) => {
  return [...result, curr*2 ];
}, [])

//indexOf with reduce
const wanted = 4;
const find = numbs.reduce((result, curr, index) => {
  if (result > -1) return result;
  return curr === wanted ? index : - 1;
}, -1)

// get sum with reduce
const sum = numbs.reduce((result, curr) => {
  return result + curr;
}, 0)
```
